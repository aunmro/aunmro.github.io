4.3.4 装箱和拆箱

using System;

class Program
{
    static void Main(string[] args)
    {
        int i = 10;
        object obj = i; //装箱：值类型封装为引用类型
        int m = (int) obj; //拆箱
        Console.WriteLine($"{i}, {obj.ToString()}, {m}");
    }
}
