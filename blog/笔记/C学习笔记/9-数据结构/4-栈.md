# 栈：数组实现

栈是一种特殊的线性表

** **特点后进先出：**，在一端进行删除和增加

```c
/*栈的数组实现*/
#include <stdbool.h>
#define STACK_MAX_SIZE 100

typedef int ElementType;

typedef struct Stack {
    ElementType data[STACK_MAX_SIZE];
    int top; /**/
} Stack;

/*初始化*/
Stack CreateStack(void) {
    Stack s;
    s.top = -1; 
    /*也可以为0，但会浪费一个空间*/
    /*栈顶元素 s.data[s.top]*/
    return s;
}

/*压栈*/
bool Push(Stack *s, ElementType data) {
    if (s->top == STACK_MAX_SIZE - 1) {
        return false; /*满栈*/
    }
    else {
        s->top++;
        s->data[s->top] = data;
        return true;
    }
}

/*出栈*/
bool Pop(Stack* s, ElementType* e) {
    if (s->top == -1) { return false; } /*空栈*/
    else {
        /*先出栈，再指针操作*/
        *e = s->data[s->top];
        s->top--;
    }
}

/*读栈顶元素*/
bool GetStackTop(Stack s, ElementType *e) {
    if (s.top == -1) { return false; } /*空栈*/
    else {
        *e = s.data[s.top];
        return true;
    }
}

/*判断栈是否为空*/
bool StackIsEmpty(Stack s) {
    if (s.top == -1) {
        return true;
    }
    else {
        return false;
    }
}

/*栈的长度*/
int StackLength(Stack s) { return s.top + 1; }
```
  
# 栈：链表实现

```c
/*链栈的实现*/

/*结构体定义*/
#include <stdbool.h>
#include <stdlib.h>
typedef int ElemType;

typedef struct LinkStackNode {
    ElemType data;
    struct LinkStackNode* next;
} LinkStackNode, LinkStack;

/*栈是否为空*/
bool LinkStackIsEmpty(LinkStack *s) {
    if (s->next == NULL) {
        return true;
    }
    else {
        return false;
    }
}

/*创建链栈*/
LinkStack* CreateLinkStack(void) {
    LinkStack *top = (LinkStackNode*)malloc(sizeof(LinkStackNode));
    top->next = NULL;
    return top;
}

/*进栈*/
void LinkStackPush(LinkStack* s,ElemType data) {
    LinkStackNode* node = (LinkStackNode*)malloc(sizeof(LinkStackNode));
    node->data = data;
    node->next = s->next;
    s->next = node;
}

/*出栈*/
bool  LinkStackPop(LinkStack* s, ElemType* output) {
    if (s->next == NULL) { return false; } // 空栈
    LinkStackNode *popNode = s->next;
    output = popNode->data;
    s->next = popNode->next;
    free(popNode);
    return true;
}
```
